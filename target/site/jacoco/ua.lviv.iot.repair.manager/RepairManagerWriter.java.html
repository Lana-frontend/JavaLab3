package ua.lviv.iot.repair.manager;

import ua.lviv.iot.repair.model.RepairInstruments;

import java.io.BufferedWriter;
import java.io.File;
import java.io.FileOutputStream;
import java.io.OutputStreamWriter;
import java.nio.charset.StandardCharsets;
import java.util.List;

public class RepairManagerWriter {

    private String filePath;

    RepairManagerWriter() {
    }

    public RepairManagerWriter(String filePath) {
        this.filePath = filePath;
    }

    public void writeToFile(List<RepairInstruments> repair) {
        File csvFile = new File(filePath + ".csv");

        try (FileOutputStream fos = new FileOutputStream(csvFile);
             OutputStreamWriter osw = new OutputStreamWriter(fos, StandardCharsets.UTF_8);
             BufferedWriter writer = new BufferedWriter(osw)
        ) {
            for (RepairInstruments repairs : repair) {
                writer.write(repairs.getHeaders());
                writer.newLine();
                writer.write(repairs.toCSV());
                writer.newLine();
            }
        } catch (Exception e) {
            e.printStackTrace();
        } finally {
            System.out.println("Writing to the file ended");
        }
    }

    public String getFilePath() {
        return filePath;
    }

    public void setFilePath(String filePath) {
        this.filePath = filePath;
    }
}
